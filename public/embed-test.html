Basic Example

This is the simplest way to embed the TakeShape Room Scanner.

HTML Code

<!-- 1. Add the embed script -->
<script src="/embed.js"></script>

<!-- 2. Add a container for the embed -->
<div id="basic-embed-container"></div>

<!-- 3. Initialize the embed -->
<script>
  TakeShapeEmbed.init({
    container: 'basic-embed-container',
    height: '600px'
  });
</script>

Custom Colors Example

You can customize the colors and appearance of the embed to match your brand.

HTML Code

<!-- 1. Add the embed script -->
<script src="/embed.js"></script>

<!-- 2. Add a container for the embed -->
<div id="custom-embed-container"></div>

<!-- 3. Initialize the embed with custom colors -->
<script>
  TakeShapeEmbed.init({
    container: 'custom-embed-container',
    primaryColor: '#4CAF50',
    secondaryColor: '#2196F3',
    backgroundColor: '#f9f9f9',
    textColor: '#333333',
    fontFamily: 'Arial, sans-serif',
    borderRadius: '12px',
    boxShadow: '0 4px 12px rgba(0, 0, 0, 0.15)',
    height: '600px'
  });
</script>

With Callbacks Example

You can add callback functions to handle events from the embed.

Callback results will appear here when the user completes the process.

HTML Code

<!-- 1. Add the embed script -->
<script src="/embed.js"></script>

<!-- 2. Add a container for the embed -->
<div id="callback-embed-container"></div>

<!-- 3. Add a container for callback results -->
<div id="callback-result"></div>

<!-- 4. Initialize the embed with callbacks -->
<script>
  TakeShapeEmbed.init({
    container: 'callback-embed-container',
    primaryColor: '#9C27B0',
    height: '600px',
    onComplete: function(data) {
      // Handle completion event
      document.getElementById('callback-result').innerHTML = 
        '<h3>Quote Accepted!</h3>' +
        '<pre>' + JSON.stringify(data, null, 2) + '</pre>';
    },
    onError: function(error) {
      // Handle error event
      document.getElementById('callback-result').innerHTML = 
        '<h3>Error Occurred</h3>' +
        '<p>' + error.message + '</p>';
    }
  });
</script>
